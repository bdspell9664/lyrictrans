# 格式转换后的歌词文件样本

## 1. AMLL 格式样本

```xml
<?xml version="1.0" encoding="UTF-8"?>
<advanced-music-lyrics xmlns="http://www.example.com/amll">
  <metadata>
    <title>示例歌曲</title>
    <artist>示例歌手</artist>
    <album>示例专辑</album>
  </metadata>
  <timeline>
    <line startTime="0.000" endTime="3.500">
      <word startTime="0.000" endTime="0.500">这</word>
      <word startTime="0.500" endTime="1.000">是</word>
      <word startTime="1.000" endTime="1.500">一</word>
      <word startTime="1.500" endTime="2.000">首</word>
      <word startTime="2.000" endTime="2.500">示</word>
      <word startTime="2.500" endTime="3.000">例</word>
      <word startTime="3.000" endTime="3.500">歌</word>
    </line>
    <line startTime="3.500" endTime="6.000">
      <word startTime="3.500" endTime="4.000">词</word>
      <word startTime="4.000" endTime="4.500">的</word>
      <word startTime="4.500" endTime="5.000">第</word>
      <word startTime="5.000" endTime="5.500">二</word>
      <word startTime="5.500" endTime="6.000">行</word>
    </line>
  </timeline>
  <styles>
    <style id="default">
      <property name="font-size">24px</property>
      <property name="color">#FFFFFF</property>
      <property name="font-family">Arial, sans-serif</property>
    </style>
    <style id="highlighted">
      <property name="color">#FF0000</property>
      <property name="font-weight">bold</property>
    </style>
  </styles>
</advanced-music-lyrics>
```

## 2. TTML 格式样本

```xml
<?xml version="1.0" encoding="UTF-8"?>
<tt xml:lang="zh-CN" xmlns="http://www.w3.org/ns/ttml" xmlns:tts="http://www.w3.org/ns/ttml#styling">
  <head>
    <styling>
      <style id="default" tts:fontSize="24px" tts:color="#FFFFFF" tts:fontFamily="Arial, sans-serif" tts:textAlign="center"/>
      <style id="highlighted" tts:color="#FF0000" tts:fontWeight="bold"/>
    </styling>
  </head>
  <body>
    <div>
      <p begin="0.000s" end="3.500s" style="default">
        <span begin="0.000s" end="0.500s" style="highlighted">这</span>
        <span begin="0.500s" end="1.000s" style="highlighted">是</span>
        <span begin="1.000s" end="1.500s" style="highlighted">一</span>
        <span begin="1.500s" end="2.000s" style="highlighted">首</span>
        <span begin="2.000s" end="2.500s" style="highlighted">示</span>
        <span begin="2.500s" end="3.000s" style="highlighted">例</span>
        <span begin="3.000s" end="3.500s" style="highlighted">歌</span>
      </p>
      <p begin="3.500s" end="6.000s" style="default">
        <span begin="3.500s" end="4.000s" style="highlighted">词</span>
        <span begin="4.000s" end="4.500s" style="highlighted">的</span>
        <span begin="4.500s" end="5.000s" style="highlighted">第</span>

      </p>
    </div>
  </body>
</tt>
```
